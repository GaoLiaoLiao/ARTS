#### MySolution:Brute Force

```Java
// time complexity: O(n*Î±(n))
// space complexity: O(n)
class Solution {
    public int[] findRedundantConnection(int[][] edges) {
        int[] res = new int[2];
        int n = edges.length;
        int[] root = new int[n + 1];
        for(int i = 1; i <= n; i++) {
            root[i] = i;
        }
        
        for(int[] e : edges) {
            int x = find(root, e[0]);
            int y = find(root, e[1]);
            if(x == y) {
                res[0] = e[0];
                res[1] = e[1];
                break;
            }
            root[y] = x;
        }
        return res;
    }
    
    public int find(int[] root, int x) {
        while(x != root[x]) {
            root[x] = root[root[x]];
            x = root[x];
        }
        return x;
    }
}
```
